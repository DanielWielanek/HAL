30 void TrackAna::ProcessEvent() {
31 fMemoryMap->PrepareMaps(fCurrentEventCollectionID);
32 CutCollection* cont = fCutContainer->GetEventCollection(fCurrentEventCollectionID);
33 for (fTrackIndex = 0; fTrackIndex < fMemoryMap->GetTemporaryTotalTracksNo(); fTrackIndex++) {
34 fCurrentTrack = fCurrentEvent->GetTrack(fTrackIndex);
35 for (int j = 0; j < cont->GetNextNo(); j++) {
36 fCurrentTrackCollectionID = cont->GetNextAddr(j);
37 if (fCutContainer->PassTrack(fCurrentTrack, fCurrentTrackCollectionID)) {
38 fMemoryMap->AddTrackToMapTrack(fCurrentEventCollectionID,
39 fCurrentTrackCollectionID,
40 fTrackIndex); // load track into memory map - may be usefull at
41 // finish event
42 ProcessTrack();
43 }
44 }
45 }
46 }

60 void TrackAna::LinkCollections() {
61 Int_t trackCollections = fCutContainer->GetTrackCollectionsNo();
62 Int_t eventCollections = fCutContainer->GetEventCollectionsNo();
63 Int_t jump = trackCollections / eventCollections;
64 if (jump == 0) {
65 Cout::PrintInfo("Too many event collections, some of them will be ignored", EInfo::kLowWarning);
66 jump++;
67 }
68 if (trackCollections != jump * eventCollections) {
69 Cout::PrintInfo("Some track collections will be ignored", EInfo::kLowWarning);
70 }
71 for (int i = 0; i < eventCollections; i++) {
72 for (int j = i * jump; j < (i + 1) * jump; j++) {
73 fCutContainer->LinkCollections(ECutUpdate::kEvent, i, ECutUpdate::kTrack, j);
74 }
75 }
76 }
