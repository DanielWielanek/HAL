18Bool_t TrackDeltaPtCut::Pass(Track* track) {
19 if (!GetMom(track)) {
20 SetValue(-1E+9);
21 return ForcedUpdate(kFALSE);
22 }
23 Double_t pT_reco = fTrackRe->GetMomentum().Pt();
24 Double_t pT_mc = fTrackIm->GetMomentum().Pt();
25 SetValue(pT_mc - pT_reco);
26 return Validate();
27}
