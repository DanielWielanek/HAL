21 void MiniEventTask::Exec(Option_t* /*opt*/) {
22 fCurrentEvent = fMemoryMap->GetTemporaryEvent();
23 fMap.Reset(fCurrentEvent->GetTotalTrackNo());
24 for (fCurrentEventCollectionID = 0; fCurrentEventCollectionID < fEventCollectionsNo; fCurrentEventCollectionID++) {
25 if (fCutContainer->PassEvent(fCurrentEvent, fCurrentEventCollectionID)) {
26 CutCollection* cont = fCutContainer->GetEventCollection(fCurrentEventCollectionID);
27 fMemoryMap->ResetTrackMaps(fCurrentEventCollectionID,
28 0); // counter always = 0 due to no buffering
29 for (int i = 0; i < fMemoryMap->GetTemporaryTotalTracksNo(); i++) {
30 for (int j = 0; j < cont->GetNextNo(); j++) {
31 fCurrentTrackCollectionID = cont->GetNextAddr(j);
32 fCurrentTrack = fCurrentEvent->GetTrack(i);
33 if (fCutContainer->PassTrack(fCurrentTrack, fCurrentTrackCollectionID)) { fMap.MarkAsGood(i); }
34 }
35 }
36 }
37 }
38 fMap.Recalculate();
39 fCurrentEvent->Compress(fMap);
40 }
