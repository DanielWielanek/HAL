30 FastHist1D::FastHist1D(TString name, TString title, Int_t nbins, Double_t min, Double_t max) : FastHist(name, title) {
31 fBinsNoX = nbins + 2;
32 if (nbins != 0) {
33 fValues = new Double_t[fBinsNoX];
34 fMinX = min;
35 fMaxX = max;

44 Int_t FastHist1D::FindBin(Double_t x) const {
